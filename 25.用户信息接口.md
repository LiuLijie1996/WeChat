+ 获取用户信息

```html
<button open-type="getUserInfo" bindgetuserinfo="userInfoHandler">获取用户信息</button>
```

```js
// 引导用户授权
getUserInfo() {
    wx.getUserInfo({
      lang: 'zh_CN',
    })
},

//同意授权后得到了用户信息
userInfoHandler(res){
    console.log(res);
},
```
